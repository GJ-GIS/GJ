ts
 // X轴 (红色)
viewer.entities.add({
  name: 'X轴',
  position: new Cesium.Cartesian3(length, 0, 0),
  label: {
    text: 'X(正东)',
    font: '20px sans-serif',
    horizontalOrigin: Cesium.HorizontalOrigin.LEFT,
    verticalOrigin: Cesium.VerticalOrigin.BOTTOM,
    pixelOffset: new Cesium.Cartesian2(15, -15),
  },
  polyline: {
    positions: [new Cesium.Cartesian3(0.01, 0, 0), new Cesium.Cartesian3(length, 0, 0)],
    width: 5,
    material: Cesium.Color.RED,
    clampToGround: false,
  },
})
// Y轴 (绿色)
viewer.entities.add({
  name: 'Y轴',
  position: new Cesium.Cartesian3(0, length, 0),
  label: {
    text: 'Y(正北)',
    font: '20px sans-serif',
    horizontalOrigin: Cesium.HorizontalOrigin.LEFT,
    verticalOrigin: Cesium.VerticalOrigin.BOTTOM,
    pixelOffset: new Cesium.Cartesian2(15, -15),
  },
  polyline: {
    positions: [new Cesium.Cartesian3(0, 0.01, 0), new Cesium.Cartesian3(0, length, 0)],
    width: 5,
    material: Cesium.Color.GREEN,
    clampToGround: false,
  },
})
// Y轴 (蓝色)
viewer.entities.add({
  name: 'Z轴',
  position: new Cesium.Cartesian3(0, 0, length),
  label: {
    text: 'Z(正上)',
    font: '20px sans-serif',
    horizontalOrigin: Cesium.HorizontalOrigin.LEFT,
    verticalOrigin: Cesium.VerticalOrigin.BOTTOM,
    pixelOffset: new Cesium.Cartesian2(15, -15),
  },
  polyline: {
    positions: [new Cesium.Cartesian3(0, 0, 0.01), new Cesium.Cartesian3(0, 0, length)],
    width: 5,
    material: Cesium.Color.BLUE,
    clampToGround: false,
  },
})
