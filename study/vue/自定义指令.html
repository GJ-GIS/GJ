方法一: 传入的数组的第二个值是一个函数 ​

传入的数组的第二个值是一个函数, 所以状态的改变并不会更改数据,因此不会重新触发渲染.所以不使用更新的钩子函数,而是使用watch监听函数的返回值,如果返回值改变,则修改style值.

vue
<script setup lang="ts">
  import { ref, computed, watch } from 'vue'

  /**
   * Implement the custom directive
   * Make sure the list item text color changes to red when the `toggleTab` is toggled
   *
   */
  let watcher
  const VActiveStyle = {
    mounted(el, binding) {
      const [style, fn] = binding.value
      const isActive = computed(() => fn())
      watcher = watch(
        isActive,
        nva => {
          for (let [key, value] of Object.entries(style)) {
            el.style[key] = isActive.value ? value : ''
          }
        },
        { immediate: true },
      )
    },
    unmounted() {
      console.log('unmo')
      watcher && watcher()
    },
    updated(el, binding) {
      console.log(binding.value)
    },
  }

  const list = ref([1, 2, 3, 4, 5, 6, 7, 8])
  const activeTab = ref(0)
  function toggleTab(index: number) {
    activeTab.value = index
  }
  let t = ref(true)
  function handleClick() {
    t.value = !t.value
  }
</script>

<template>
  <button @click="handleClick">{{ t }}</button>
  <ul v-if="t">
    <li v-for="(item, index) in list" :key="index" v-active-style="[{ color: 'red' }, () => activeTab === index]" @click="toggleTab(index)">
      {{ item }}
    </li>
  </ul>
</template>

方法二: 使用自定义指令的updated. ​

传入的数组的第二个值不能再为一个函数, 而应该直接是状态,一旦状态发生改变.就会触发组件和自定义指令的更新事件.容易触发过于频繁,消耗性能.

vue
<script setup lang="ts">
  import { ref, watchEffect } from 'vue'

  const bl = ref(false)
  const handleClick = () => {
    bl.value = !bl.value
  }
  function uppercase(value) {
    return value.charAt(0).toUpperCase() + value.slice(1)
  }

  const vUppercase = {
    mounted(el, binding) {
      const bValue = binding.value
      el.addEventListener('input', () => {
        console.log(binding)
        el.value = uppercase(el.value)
      })
    },
    updated(el, binding) {
      console.log(binding)
    },
  }

  /**
   * Implement a custom directive
   * Create a two-way binding on a form input element
   *
   */
  const VOhModel = {
    mounted(el, binding) {
      let inputValue = binding.value
      console.log(binding)
      el.value = inputValue
      el.addEventListener('input', () => {
        value.value = el.value
      })
    },
    updated(el, binding) {},
    unMounted(el, binding) {},
  }

  const value = ref('Hello Vue.js')
</script>

<template>
  <button style="display: block" @click="handleClick">{{ bl }}</button>
  <input v-uppercase="bl" v-oh-model="value" type="text" />
  <p>{{ value }}</p>
</template>
