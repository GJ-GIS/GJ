// box的长度为100000米

const position = Cesium.Cartesian3.fromDegrees(116.39, 39.9, 0.5 * boxLength)

const modelMatrix = Cesium.Transforms.eastNorthUpToFixedFrame(position)

1. 定义位置数组 ​

目的：构造模型形状; boxVertex 和 positions

 立方体                        坐标系
    v6----- v5                   z
   /|      /|                    | /y
  v1------v0|                    |/
  | |     | |                    o------x
  | |v7---|-|v4
  |/      |/
  v2------v3
ts
// 定义立方体顶点位置
const vertices = [
  [0.5, -0.5, 0.5], // v0
  [-0.5, -0.5, 0.5], // v1
  [-0.5, -0.5, -0.5], // v2
  [0.5, -0.5, -0.5], // v3
  [0.5, 0.5, -0.5], // v4
  [0.5, 0.5, 0.5], // v5
  [-0.5, 0.5, 0.5], // v6
  [-0.5, 0.5, -0.5], // v7
]

// 1-定义立方体的顶点顺序
const rawVertex = [
  // 下 -z
  ...vertices[2],
  ...vertices[3],
  ...vertices[4],
  ...vertices[7],
  // 前 -y
  ...vertices[2],
  ...vertices[3],
  ...vertices[0],
  ...vertices[1],
  // 后 +y
  ...vertices[4],
  ...vertices[7],
  ...vertices[6],
  ...vertices[5],
  // 左 -x
  ...vertices[7],
  ...vertices[2],
  ...vertices[1],
  ...vertices[6],
  // 右 +x
  ...vertices[3],
  ...vertices[4],
  ...vertices[5],
  ...vertices[0],
  // 上 +z
  ...vertices[1],
  ...vertices[0],
  ...vertices[5],
  ...vertices[6],
]
// 将顶点坐标乘以立方体长度
const boxVertex = rawVertex.map(v => v * boxLength)
// Float64Array: 8 字节 (64 位);15-16 位有效数字(总位数)
// const positions = new Float64Array(rawVertex)
// Float32Array: 4 字节 (32 位);7 位有效数字(总位数)
const positions = new Float32Array(boxVertex)

5. 创建自定义几何体 ​

ts
const primitive = new Cesium.Primitive({
  geometryInstances: new Cesium.GeometryInstance({
    geometry: new Cesium.Geometry({
      attributes: {
        position: new Cesium.GeometryAttribute({
          componentDatatype: Cesium.ComponentDatatype.DOUBLE,
          componentsPerAttribute: 3,
          values: positions,
        }),
        normal: new Cesium.GeometryAttribute({
          componentDatatype: Cesium.ComponentDatatype.FLOAT,
          componentsPerAttribute: 3,
          values: normals,
        }),
        st: new Cesium.GeometryAttribute({
          componentDatatype: Cesium.ComponentDatatype.FLOAT,
          componentsPerAttribute: 2,
          values: sts,
        }),
        bitangent: undefined,
        color: undefined,
        tangent: undefined,
      },
      indices,
      primitiveType: Cesium.PrimitiveType.TRIANGLES,
      boundingSphere: Cesium.BoundingSphere.fromVertices(boxVertex),
    }),
  }),
  appearance: new Cesium.MaterialAppearance({
    material: Cesium.Material.fromType('Image', {
      image: moonImage,
    }),
    closed: true,
  }),
  modelMatrix: modelMatrix, // 应用模型矩阵
  // 确定是异步创建基元，还是在准备就绪之前阻止基元。false：阻止
  asynchronous: true,
})
